class RandomUser {
  final String fullName;
  final String email;
  final String image;
  final String street;
  final String city;
  final String state;
  final String country;

  RandomUser({
    required this.fullName,
    required this.email,
    required this.image,
    required this.street,
    required this.city,
    required this.state,
    required this.country,
  });

  factory RandomUser.fromJson(Map<String, dynamic> json) {
    final name = json['name'];
    final location = json['location'];

    return RandomUser(
      fullName: "${name['first']} ${name['last']}",
      email: json['email'],
      image: json['picture']['large'],
      street: location['street']['name'],
      city: location['city'],
      state: location['state'],
      country: location['country'],
    );
  }
}

import 'package:http/http.dart' as http;
import 'dart:convert';
import '../model/random_user.dart';

// دالة لجلب المستخدمين العشوائيين
Future<List<RandomUser>> featchedUsers() async {
  final response = await http.get(Uri.parse('https://randomuser.me/api/?results=5'));

  if (response.statusCode == 200) {
    List<dynamic> body = jsonDecode(response.body)['results'];
    return body.map((json) => RandomUser.fromJson(json)).toList();
  } else {
    throw Exception('Error In Fetching Users');
  }
}

import 'package:flutter/material.dart';
import '../data/helper.dart';
import '../model/random_user.dart';
import 'UserDetails.dart';

class WidgetScreen extends StatefulWidget {
  const WidgetScreen({super.key});

  @override
  State<WidgetScreen> createState() => _WidgetScreenState();
}

class _WidgetScreenState extends State<WidgetScreen> {
  late Future<List<RandomUser>> _futureUsers;

  @override
  void initState() {
    super.initState();
    _futureUsers = featchedUsers();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('Random Users')),
      body: FutureBuilder<List<RandomUser>>(
        future: _futureUsers,
        builder: (context, snapshot) {
          if (snapshot.connectionState == ConnectionState.waiting) {
            return Center(child: CircularProgressIndicator());
          } else if (snapshot.hasError) {
            return Center(child: Text('Error: ${snapshot.error}'));
          } else if (snapshot.hasData) {
            final users = snapshot.data!;
            return ListView.builder(
              itemCount: users.length,
              itemBuilder: (context, index) {
                final user = users[index];
                return Card(
                  margin: EdgeInsets.symmetric(horizontal: 12, vertical: 8),
                  child: ListTile(
                    leading: CircleAvatar(
                      backgroundImage: NetworkImage(user.image),
                      radius: 30,
                    ),
                    title: Text(user.fullName),
                    subtitle: Text(user.email),
                    onTap: () {
                      Navigator.push(
                        context,
                        MaterialPageRoute(builder: (_) => UserDetails(user: user)),
                      );
                    },
                  ),
                );
              },
            );
          } else {
            return Center(child: Text('No data found'));
          }
        },
      ),
      floatingActionButton: FloatingActionButton(
        onPressed: () {
          setState(() {
            _futureUsers = featchedUsers();
          });
        },
        child: Icon(Icons.refresh),
      ),
    );
  }
}

import 'package:flutter/material.dart';
import 'WidgetScreen.dart';
import 'registration_screen.dart';

class LoginScreen extends StatefulWidget {
  @override
  State<LoginScreen> createState() => _LoginScreenState();
}

class _LoginScreenState extends State<LoginScreen> {
  final TextEditingController emailController = TextEditingController();
  final TextEditingController passwordController = TextEditingController();

  String? errorMessage;

  void loginUser() {
    const allowedEmail = "admin@gmail.com";
    const allowedPassword = "123456";

    if (emailController.text.trim() == allowedEmail &&
        passwordController.text.trim() == allowedPassword) {
      Navigator.pushReplacement(
        context,
        MaterialPageRoute(builder: (_) => WidgetScreen()),
      );
    } else {
      setState(() {
        errorMessage = "Invalid email or password";
      });
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Login")),
      body: Padding(
        padding: const EdgeInsets.all(20),
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            TextField(
              controller: emailController,
              decoration: InputDecoration(labelText: "Email", border: OutlineInputBorder()),
            ),
            SizedBox(height: 20),
            TextField(
              controller: passwordController,
              decoration: InputDecoration(labelText: "Password", border: OutlineInputBorder()),
              obscureText: true,
            ),
            if (errorMessage != null) ...[
              SizedBox(height: 15),
              Text(errorMessage!, style: TextStyle(color: Colors.red, fontWeight: FontWeight.bold)),
            ],
            SizedBox(height: 25),
            ElevatedButton(onPressed: loginUser, child: Text("Login")),
            TextButton(
              onPressed: () {
                Navigator.push(context, MaterialPageRoute(builder: (_) => RegistrationScreen()));
              },
              child: Text("Don’t have an account? Register"),
            ),
          ],
        ),
      ),
    );
  }
}
